systemctl status ssh - проверить статус ssh сервера
ssh root@<IP_ВМ2> - доступ с помощью ssh  по указанному айпи 
ssh -p(порт)root@<IP_ВМ2> - доступ с нестандартным портом 
/etc/ssh/sshd_config - здесь можно поправить порт для подключения(незабудь презапустить сервер)
systemctl restart ssh - рестарт ssh сервера для того что бы настройки применились
scp -P 3838 -r /cdmounted/* root@192.168.0.2:/distr/ - копирование ОТКУДА в КУДА (с использованием нестандартного порта)
ss -tulnp | grep sshd - проверить на каком порту слушает sshd
